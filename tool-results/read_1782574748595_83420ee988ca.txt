    52→// ─────────────────────────────────────────────────────────────────────────────
    53→// Robust JSON extraction from an LLM response
    54→// ─────────────────────────────────────────────────────────────────────────────
    55→function extractJson(text: string): unknown {
    56→  if (!text) throw new Error('empty LLM response')
    57→  // strip code fences
    58→  let t = text.trim()
    59→  const fence = t.match(/```(?:json)?\s*([\s\S]*?)```/i)
    60→  if (fence) t = fence[1].trim()
    61→  // find outermost braces
    62→  const start = t.indexOf('{')
    63→  const end = t.lastIndexOf('}')
    64→  if (start !== -1 && end !== -1 && end > start) t = t.slice(start, end + 1)
    65→  return JSON.parse(t)
    66→}

   177→// ─────────────────────────────────────────────────────────────────────────────
   178→// LIGHTWEIGHT RETRIEVAL (keyword-overlap scoring — no external vector DB needed)
   179→// ─────────────────────────────────────────────────────────────────────────────
   180→function tokenize(s: string): string[] {
   181→  return (s.toLowerCase().match(/[a-z0-9]{2,}/g) ?? []).filter(
   182→    (w) => !STOP.has(w) && w.length > 2,
   183→  )
   184→}
   185→const STOP = new Set([
   186→  'the','and','for','that','this','with','from','your','you','what','why','how',
   187→  'was','were','are','did','does','have','had','they','their','them','his','her',
   188→  'she','him','his','about','into','than','then','when','which','who','whom',
   189→  'not','but','but','are','can','could','would','should','will','there','here',
   190→])
   191→
   192→export function retrieveMemories(question: string, memories: { id: string; title: string; content: string; keywords: string }[], k = 6) {
   193→  const qTokens = new Set(tokenize(question))
   194→  if (qTokens.size === 0) return memories.slice(0, k)
   195→  const scored = memories.map((m) => {
   196→    const hay = tokenize(`${m.title} ${m.content} ${m.keywords}`)
   197→    const counts = new Map<string, number>()
   198→    for (const t of hay) counts.set(t, (counts.get(t) ?? 0) + 1)
   199→    let score = 0
   200→    for (const t of qTokens) score += counts.get(t) ?? 0
   201→    // small boost for keyword exact matches
   202→    const kws = m.keywords.toLowerCase().split(',').map((s) => s.trim()).filter(Boolean)
   203→    for (const k of kws) if (qTokens.has(k)) score += 3
   204→    return { m, score }
   205→  })
   206→  scored.sort((a, b) => b.score - a.score)
   207→  return scored.filter((s) => s.score > 0).slice(0, k).map((s) => s.m)
   208→}
